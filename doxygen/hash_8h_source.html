406 template<typename T, typename U>
407 inline constexpr T
408 string_hash(U && str, const std::size_t n) {
409  if(n == 0)
410  return 0;
411 
412  // String-to-integer hash function, based on prime numbers.
413  // References:
414  // https://stackoverflow.com/questions/8317508/hash-function-for-a-string
415  // https://planetmath.org/goodhashtableprimes
416 
417  const T P = 3145739; // prime
418  const T Q = 6291469; // prime, a bit less than 2x the first
419 
420  T h = 37; // prime
421  for(std::size_t i = 0; i < n; ++i)
422  h = (h * P) ^ (str[i] * Q);
423  return h;
424 } // string_hash
