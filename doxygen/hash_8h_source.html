413 template<typename T, typename U>
414 inline constexpr T
415 string_hash(U && str, const std::size_t n) {
416  if(n == 0)
417  return 0;
418 
419  // String-to-integer hash function, based on prime numbers.
420  // References:
421  // https://stackoverflow.com/questions/8317508/hash-function-for-a-string
422  // https://planetmath.org/goodhashtableprimes
423 
424  const T P = 3145739; // prime
425  const T Q = 6291469; // prime, a bit less than 2x the first
426 
427  T h = 37; // prime
428  for(std::size_t i = 0; i < n; ++i)
429  h = (h * P) ^ (str[i] * Q);
430  return h;
431 } // string_hash
