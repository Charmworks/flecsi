289  template<size_t NAMESPACE, typename OBJECT_TYPE, typename... ARGS>
290  OBJECT_TYPE * add_global_object(size_t index, ARGS &&... args) {
291  size_t KEY = NAMESPACE ^ index;
292 
293  flog_assert(
294  task_depth() == 0, "you cannot add global objects from within a task");
295 
296  flog_assert(
297  global_object_registry_.find(KEY) == global_object_registry_.end(),
298  "global key already exists");
299 
300  auto ptr = new OBJECT_TYPE(std::forward<ARGS>(args)...);
301 
302  flog(internal) << "Adding global object" << std::endl
303  << "\tindex: " << index << std::endl
304  << "\thash: " << NAMESPACE << std::endl
305  << "\ttype: " << utils::demangle(typeid(OBJECT_TYPE).name())
306  << std::endl
307  << "\taddress: " << ptr << std::endl;
308 
309  std::get<0>(global_object_registry_[KEY]) =
310  reinterpret_cast<uintptr_t>(ptr);
311 
312  using wrapper_t = global_object_wrapper_u<OBJECT_TYPE>;
313  std::get<1>(global_object_registry_[KEY]) = &wrapper_t::cleanup;
314 
315  return ptr;
