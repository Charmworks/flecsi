269  template<size_t NAMESPACE, typename OBJECT_TYPE, typename... ARGS>
270  OBJECT_TYPE * add_global_object(size_t index, ARGS &&... args) {
271  size_t KEY = NAMESPACE ^ index;
272 
273  flog_assert(
274  task_depth() == 0, "you cannot add global objects from within a task");
275 
276  flog_assert(
277  global_object_registry_.find(KEY) == global_object_registry_.end(),
278  "global key already exists");
279 
280  auto ptr = new OBJECT_TYPE(std::forward<ARGS>(args)...);
281 
282  flog(internal) << "Adding global object" << std::endl
283  << "\tindex: " << index << std::endl
284  << "\thash: " << NAMESPACE << std::endl
285  << "\ttype: " << utils::demangle(typeid(OBJECT_TYPE).name())
286  << std::endl
287  << "\taddress: " << ptr << std::endl;
288 
289  std::get<0>(global_object_registry_[KEY]) =
290  reinterpret_cast<uintptr_t>(ptr);
291 
292  using wrapper_t = global_object_wrapper_u<OBJECT_TYPE>;
293  std::get<1>(global_object_registry_[KEY]) = &wrapper_t::cleanup;
294 
295  return ptr;
