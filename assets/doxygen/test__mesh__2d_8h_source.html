263 void
264 initialize_mesh(data_client_handle__<test_mesh_2d_t, wo> mesh) {
265  auto & context = execution::context_t::instance();
266 
267  auto & vertex_map{context.index_map(index_spaces::vertices)};
268  auto & reverse_vertex_map{context.reverse_index_map(index_spaces::vertices)};
269  auto & cell_map{context.index_map(index_spaces::cells)};
270 
271  std::vector<vertex_t *> vertices;
272 
273 #ifdef FLECSI_8_8_MESH
274  const size_t width{8};
275 #else
276  const size_t width{16};
277 #endif
278 
279  // make vertices
280  for (auto & vm : vertex_map) {
281  const size_t mid{vm.second};
282  const size_t row{mid / (width + 1)};
283  const size_t column{mid % (width + 1)};
284  // printf("vertex %lu: (%lu, %lu)\n", mid, row, column);
285  point_t point({{(double)row, (double)column}});
286  index_t index({{row, column}});
287 
288  vertices.push_back(mesh.make<vertex_t>(point, index));
289  } // for
290 
291  // make cells
292  std::unordered_map<size_t, cell_t *> cells_vs_ids;
293 
294  for (auto & cm : cell_map) {
295  const size_t mid{cm.second};
296 
297  const size_t row{mid / width};
298  const size_t column{mid % width};
299 
300  const size_t v0{(column) + (row) * (width + 1)};
301  const size_t v1{(column + 1) + (row) * (width + 1)};
302  const size_t v2{(column + 1) + (row + 1) * (width + 1)};
303  const size_t v3{(column) + (row + 1) * (width + 1)};
304 
305  const size_t lv0{reverse_vertex_map[v0]};
306  const size_t lv1{reverse_vertex_map[v1]};
307  const size_t lv2{reverse_vertex_map[v2]};
308  const size_t lv3{reverse_vertex_map[v3]};
309 
310  auto c{mesh.make<cell_t>(index_t{{row, column}})};
311  cells_vs_ids[mid] = c;
312  mesh.init_cell<0>(
313  c, {vertices[lv0], vertices[lv1], vertices[lv2], vertices[lv3]});
314  } // for
315 
316  mesh.init<0>();
317 
318  // set neighbor cells on cells that are owned by this rank (exclusive and
319  // shared), all the neighbors also exist on this rank, although some of them
320  // are ghost cells
321  auto & cell_coloring{context.coloring(index_spaces::cells)};
322 
323  auto exclusive_and_shared = cell_coloring.exclusive;
324  exclusive_and_shared.insert(
325  cell_coloring.shared.begin(), cell_coloring.shared.end());
326 
327  for (auto c : exclusive_and_shared) {
328  auto this_cell = cells_vs_ids.at(c.id);
329  size_t row = this_cell->index()[0];
330  size_t column = this_cell->index()[1];
331 
332  // left
333  if (column > 0) {
334  size_t left_id = (column - 1) + row * width;
335  this_cell->set_neighbor(left_, cells_vs_ids.at(left_id));
336  }
337  // right
338  if (column < width - 1) {
339  size_t right_id = (column + 1) + row * width;
340  this_cell->set_neighbor(right_, cells_vs_ids.at(right_id));
341  }
342  // below
343  if (row > 0) {
344  size_t below_id = column + (row - 1) * width;
345  this_cell->set_neighbor(below_, cells_vs_ids.at(below_id));
346  }
347  // above
348  if (row < width - 1) {
349  size_t above_id = column + (row + 1) * width;
350  this_cell->set_neighbor(above_, cells_vs_ids.at(above_id));
351  }
352  } // for
353 
354 } // initizlize_mesh
