289  metadata.origin_types.insert({ghost_owner, origin_type});

328  // Each shared and ghost cells element is an array of max_entries_per_index
329  // of entry_value_t

331  MPI_Type_contiguous(sizeof(data::sparse_entry_value__<T>) * 5,
332  MPI_BYTE, &shared_ghost_type);
333  MPI_Type_commit(&shared_ghost_type);
334  for (auto ghost_owner : coloring_info.ghost_owners) {
335  MPI_Datatype origin_type;
336  MPI_Datatype target_type;
337 
338  MPI_Type_indexed(metadata.compact_origin_lengs[ghost_owner].size(),
339  metadata.compact_origin_lengs[ghost_owner].data(),
340  metadata.compact_origin_disps[ghost_owner].data(),
341  //flecsi::coloring::mpi_typetraits__<T>::type(),
342  shared_ghost_type,
343  &origin_type);
344  MPI_Type_commit(&origin_type);
345  metadata.origin_types.insert({ghost_owner, origin_type});
346 
347  MPI_Type_indexed(metadata.compact_target_lengs[ghost_owner].size(),
348  metadata.compact_target_lengs[ghost_owner].data(),
349  metadata.compact_target_disps[ghost_owner].data(),
350  //flecsi::coloring::mpi_typetraits__<T>::type(),

361  auto data = sparse_field_data[fid].entries.data();
362  auto shared_data = data + sparse_field_data[fid].reserve *
363  sparse_field_data[fid].max_entries_per_index *
364  entry_value_size;

378  void register_field_metadata_(
379  MD& metadata,
380  const field_id_t fid,
381  const coloring_info_t& coloring_info,
382  const index_coloring_t& index_coloring,
383  std::map<int, std::vector<int>>& compact_origin_lengs,
384  std::map<int, std::vector<int>>& compact_origin_disps,
385  std::map<int, std::vector<int>>& compact_target_lengs,
386  std::map<int, std::vector<int>>& compact_target_disps
387  )
388  {
389  // The group for MPI_Win_post are the "origin" processes, i.e.
390  // the peer processes calling MPI_Get to get our shared cells. Thus
391  // granting access of local window to these processes. This is the set
392  // coloring_info_t::shared_users
393  // On the other hand, the group for MPI_Win_start are the 'target'
394  // processes, i.e. the peer processes this rank is going to get ghost
395  // cells from. This is the set coloring_info_t::ghost_owners.
396  // Since both shared_users and ghost_owners are std::set, we have copy
397  // them to std::vector be passed to MPI.
398  std::vector<int> shared_users(coloring_info.shared_users.begin(),
399  coloring_info.shared_users.end());
400  std::vector<int> ghost_owners(coloring_info.ghost_owners.begin(),
401  coloring_info.ghost_owners.end());
402 
403  MPI_Group comm_grp;
404  MPI_Comm_group(MPI_COMM_WORLD, &comm_grp);
405 
406  MPI_Group_incl(comm_grp, shared_users.size(),
407  shared_users.data(), &metadata.shared_users_grp);
408  MPI_Group_incl(comm_grp, ghost_owners.size(),
409  ghost_owners.data(), &metadata.ghost_owners_grp);
410 
411  std::map<int, std::vector<int>> origin_lens;
412  std::map<int, std::vector<int>> origin_disps;
413  std::map<int, std::vector<int>> target_lens;
414  std::map<int, std::vector<int>> target_disps;
415 
416  for (auto ghost_owner : ghost_owners) {
417  origin_lens.insert({ghost_owner, {}});
418  origin_disps.insert({ghost_owner, {}});
419  target_lens.insert({ghost_owner, {}});
420  target_disps.insert({ghost_owner, {}});
421  }
422 
423  int origin_index = 0;
424  for (const auto& ghost : index_coloring.ghost) {
425  origin_lens[ghost.rank].push_back(1);
426  origin_disps[ghost.rank].push_back(origin_index++);
427  target_lens[ghost.rank].push_back(1);
428  target_disps[ghost.rank].push_back(ghost.offset);
429  }
430 
431  int my_color;
432  MPI_Comm_rank(MPI_COMM_WORLD, &my_color);
433 
434  if (my_color == 0) {
435  for (auto ghost_owner : ghost_owners) {
436  std::cout << "ghost owner: " << ghost_owner << std::endl;
437  std::cout << "\torigin length: ";
438  for (auto len : origin_lens[ghost_owner]) {
439  std::cout << len << " ";
440  }
441  std::cout << std::endl;
442  std::cout << "\torigin disp: ";
443  for (auto len : origin_disps[ghost_owner]) {
444  std::cout << len << " ";
445  }
446  std::cout << std::endl;
447  std::cout << "\ttarget length: ";
448  for (auto len : target_lens[ghost_owner]) {
449  std::cout << len << " ";
450  }
451  std::cout << std::endl;
452 
453  std::cout << "\ttarget disp: ";
454  for (auto len : target_disps[ghost_owner]) {
455  std::cout << len << " ";
456  }
457  std::cout << std::endl;
458 
459  }
460  }
461 
462  for (auto ghost_owner : ghost_owners) {
463  if (origin_disps.size() == 0)
464  break;
465 
466  int count = 0;
467  compact_origin_lengs[ghost_owner].push_back(1);
468  compact_origin_disps[ghost_owner].push_back(
469  origin_disps[ghost_owner][0]);
470 
471  for (int i = 1; i < origin_disps[ghost_owner].size(); i++) {
472  if (origin_disps[ghost_owner][i] - origin_disps[ghost_owner][i - 1] ==
473  1) {
474  compact_origin_lengs[ghost_owner].back() = compact_origin_lengs[ghost_owner].back() + 1;
475  } else {
476  compact_origin_lengs[ghost_owner].push_back(1);
477  compact_origin_disps[ghost_owner].push_back(origin_disps[ghost_owner][i]);
478  }
479  }
480  }
481 
482  if (my_color == 0) {
483  for (auto ghost_owner : ghost_owners) {
484  std::cout << "ghost owner: " << ghost_owner << std::endl;
485  std::cout << "source compacted length: ";
486  for (auto len : compact_origin_lengs[ghost_owner]) {
487  std::cout << len << " ";
488  }
489  std::cout << std::endl;
490  std::cout << "source compacted disps: ";
491  for (auto disp : compact_origin_disps[ghost_owner]) {
492  std::cout << disp << " ";
493  }
494  std::cout << std::endl;
495  }
496  }
497 
498  for (auto ghost_owner : ghost_owners) {
499  if (target_disps.size() == 0)
500  break;
501

503  compact_target_lengs[ghost_owner].push_back(1);
504  compact_target_disps[ghost_owner].push_back(target_disps[ghost_owner][0]);
