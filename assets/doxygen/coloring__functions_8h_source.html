79  flecsi::topology::entity_referencers<cell_dim, ENTITY_DIM>(md, i);

89  // Iterate the direct referencers to assign entity ownership.
90  for(auto c: referencers) {
91 
92  // Check the remote info map to see if this cell is
93  // off-color. If it is, compare it's rank for
94  // the ownership logic below.
95  if(remote_info_map.find(c) != remote_info_map.end()) {
96  min_rank = std::min(min_rank, remote_info_map.at(c).rank);
97  shared_entities.insert(remote_info_map.at(c).rank);
98  }

101  // it is a local cell.
102

104  min_rank = std::min(min_rank, size_t(rank));
105 
106  // If the local cell is shared, we need to add all of
107  // the ranks that reference it.
108  if(shared_cells_map.find(c) != shared_cells_map.end())
109  shared_entities.insert(

111  shared_cells_map.at(c).shared.end()
112  );
113  } // if
114 
115  // Iterate through the closure intersection map to see if the
116  // indirect reference is part of another rank's closure, i.e.,
117  // that it is an indirect dependency.
118  for(auto ci: closure_intersection_map)
119  if(ci.second.find(c) != ci.second.end())
120  shared_entities.insert(ci.first);
121  } // for
122 
123  if(min_rank == rank) {
124  // This is a entity that belongs to our rank.
125  auto entry = entity_info_t(i, rank, offset++, shared_entities);
126  entity_info.insert( entry );
127  }
128  else {

131  } // if
132  } // for

140  // if it belongs to other colors, its a shared entity
141  if(i.shared.size()) {
142  entities.shared.insert(i);
143  // Collect all colors with whom we require communication
144  // to send shared information.
145  entity_color_info.shared_users =

147  }
148  // otherwise, its exclusive
149  else
150  entities.exclusive.insert(i);
151  } // for
152 
153  {

156

160  // Collect all colors with whom we require communication
161  // to receive ghost information.
162  entity_color_info.ghost_owners.insert(r);
163  // increment counter
164  ++offset;
165  } // for
166 
167  ++r;
168  } // for
169  } // scope

173  clog_container_one(

176  entities.exclusive,

178  );
179  clog_container_one(

181  );
182  clog_container_one(

184  );
