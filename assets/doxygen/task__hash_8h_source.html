80 // \struct task_hash_t execution/common/task_hash.h
81 //
82 // The task_hash_t type provides a aggregate hashing type that combines
83 // the task address and an enumerated processor type \ref processor_t
84 // into a single key. There is currently space for 256 unique processor
85 // types (8 bits).
86 //
87 // \note There is some potential for this hashing strategy to not
88 // be robust. In particular, we are implicitly making the assumption
89 // that the 8 most significant bits are not the only part of the
90 // task address that uniquely identifies it from other tasks.
92 struct task_hash_t
93 {
94  using key_t = task_hash_key_t::tuple_key_t;
95 
97  // Make a key from a task address (uintptr_t) and a processor type.
98  //
99  // \return a std::pair<uintptr_t, processor_t>.
101  static
102  task_hash_key_t
103  make_key(
104  uintptr_t address,
105  processor_t processor,
106  launch_t launch_type
107  )
108  {
109  return std::make_tuple(address, processor, launch_type);
110  } // make_key

113  // Custom hashing function for task registry.
114  //
115  // The key is a std::tuple<uintptr_t, processor_t, launch_t>. We assume
116  // that the task address is unique even after shifting away the 8 least
117  // significant bits. There are 4 bits for the processor type, and 4 bits
118  // for the launch type, i.e., there can be 16 unique processor types and
119  // 16 unique launch types.
120  //
121  // \param k A hash key reference. This method is required for using
122  // this hash type for std::map and std::unordered_map.
124  std::size_t
125  operator () (
126  const task_hash_key_t & key
127  )
128  const
129  {
130  // FIXME: We may be able to use a better hash strategy
131  //return (std::get<0>(k) << 8) ^ (std::get<1>(k) << 4) ^ std::get<2>(k);
132  return (key.address() << 8) ^ (key.processor() << 4) ^ key.launch();
133  } // operator ()
134
