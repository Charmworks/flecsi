132 void ExtLegionHandshake::legion_init(void)
133 {
134  CHECK_PTHREAD( pthread_mutex_lock(&sync_mutex) );
135 
136  if(!legion_count) {
137  // first legion thread creates the events/queues
138  // for later synchronization, then arrive at initialization barrier
139  ext_queue = new UserEvent[ext_queue_depth];
140  for(int i = 0; i < ext_queue_depth; i++)
141  ext_queue[i] = ((i || (state == IN_EXT)) ?
142  UserEvent::create_user_event() :
143  UserEvent());
144 
145  legion_queue = new UserEvent[legion_queue_depth];
146  for(int i = 0; i < legion_queue_depth; i++)
147  legion_queue[i] = ((i || (state == IN_LEGION)) ?
148  UserEvent::create_user_event() :
149  UserEvent());
150  }
151 
152  //printf("handshake %p: legion init - counts = L=%d, E=%d\n",
153  // this, legion_count, ext_count);
154  legion_count++;
155

157  // no external threads have arrived, so sleep until one does

172 void ExtLegionHandshake::ext_handoff_to_legion(void)
173 {
174  assert(state == IN_EXT);
175 
176  // we'll trigger the first event in the ext queue, but first,
177  // create a new event for the legion queue
178  // and shift it onto the end
179  assert(legion_queue[0].has_triggered());
180  for(int i = 1; i < legion_queue_depth; i++)
181  legion_queue[i - 1] = legion_queue[i];
182  legion_queue[legion_queue_depth - 1] = UserEvent::create_user_event();
183 
184  state = IN_LEGION;
185  ext_queue[0].trigger();
186 }
