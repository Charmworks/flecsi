172

175  auto & meta_data = sub_map.second;
176 
177  // loop over each element in the namespace
178  auto itr = meta_data.begin();
179  while ( itr != meta_data.end() ) {
180  // get the meta data key and label
181  auto & meta_data_key = itr->first;
182  auto & label = itr->second.label;
183  // now build the hash for this label

185  auto hash = key_hash ^ runtime_namespace;

841  "key already exists");
