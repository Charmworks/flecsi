340  void
341  fill_adjacency(
342  size_t from_index_space,
343  size_t to_index_space,
344  size_t color,
345  size_t size,
346  uint64_t* offsets,
347  uint64_t* indices
348  )
349  {
350  using namespace std;
351 
352  using namespace Legion;
353  using namespace LegionRuntime;
354  using namespace Arrays;
355 
356  using namespace execution;
357 
358  context_t & context = context_t::instance();
359 
360  auto itr = adjacency_map_.find({from_index_space, to_index_space});
361  clog_assert(itr != adjacency_map_.end(), "invalid adjacency");
362  adjacency_t& c = itr->second;
363  index_space_t& iis = index_space_map_[from_index_space];
364 
365  IndexSpace is = h.create_index_space(0, size - 1);
366  FieldSpace fs = h.create_field_space();
367  FieldAllocator a = h.create_field_allocator(fs);
368 
369  auto adjacency_offset_fid =
370  FieldID(internal_field::adjacency_offset);
371 
372  auto adjacency_index_fid =
373  FieldID(internal_field::adjacency_index);
374 
375  a.allocate_field(sizeof(uint64_t), adjacency_offset_fid);
376  a.allocate_field(sizeof(uint64_t), adjacency_index_fid);
377 
378  LogicalRegion lr = h.create_logical_region(is, fs);
379 
380  RegionRequirement rr(lr, WRITE_DISCARD, EXCLUSIVE, lr);
381  rr.add_field(adjacency_offset_fid);
382  rr.add_field(adjacency_index_fid);
383  InlineLauncher il(rr);
384 
385  PhysicalRegion pr = runtime_->map_region(ctx_, il);
386  pr.wait_until_valid();
387 
388  uint64_t* dst_offsets;
389  h.get_buffer(pr, dst_offsets, adjacency_offset_fid);
390 
391  uint64_t* dst_indices;
392  h.get_buffer(pr, dst_indices, adjacency_index_fid);
393 
394  std::memcpy(dst_offsets, offsets, sizeof(uint64_t) * size);
395  std::memcpy(dst_indices, indices, sizeof(uint64_t) * size);
396 
397  runtime_->unmap_region(ctx_, pr);
398 
399  auto fill_adjacency_tid =
400  context.task_id<__flecsi_internal_task_key(fill_adjacency_task)>();
401 
402  FieldID adjacency_fid =
403  context.adjacency_fid(from_index_space, to_index_space);
404 
405  auto p = std::make_tuple(from_index_space, to_index_space, size);
406 
407  TaskLauncher l(fill_adjacency_tid, TaskArgument(&p, sizeof(p)));
408 
409  LogicalPartition color_conn_lp =
410  runtime_->get_logical_partition(ctx_,
411  c.logical_region, c.index_partition);
412 
413  LogicalRegion color_conn_lr =
414  runtime_->get_logical_subregion_by_color(ctx_, color_conn_lp, color);
415 
416  RegionRequirement rr1(color_conn_lr, WRITE_DISCARD, SIMULTANEOUS,
417  c.logical_region);
418  rr1.add_field(adjacency_index_fid);
419  l.add_region_requirement(rr1);
420 
421  LogicalPartition color_lp =
422  runtime_->get_logical_partition(ctx_,
423  iis.logical_region, iis.index_partition);
424 
425  LogicalRegion color_lr =
426  runtime_->get_logical_subregion_by_color(ctx_, color_lp, color);
427 
428  RegionRequirement rr2(color_lr, WRITE_DISCARD, SIMULTANEOUS,
429  iis.logical_region);
430  rr2.add_field(adjacency_fid);
431  l.add_region_requirement(rr2);
432 
433  RegionRequirement rr3(lr, READ_ONLY, SIMULTANEOUS, lr);
434  rr3.add_field(adjacency_offset_fid);
435  rr3.add_field(adjacency_index_fid);
436  l.add_region_requirement(rr3);
437 
438  MustEpochLauncher must_epoch_launcher;
439  DomainPoint point(color);
440  must_epoch_launcher.add_single_task(point, l);
441 
442  auto future = runtime_->execute_must_epoch(ctx_, must_epoch_launcher);
443  future.wait_all_results();
444 
445  runtime_->destroy_index_space(ctx_, is);
446  runtime_->destroy_field_space(ctx_, fs);
447  runtime_->destroy_logical_region(ctx_, lr);
448  }
