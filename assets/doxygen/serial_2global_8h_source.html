435  decltype(auto)
436  get_accessors(
437  const data_client_t & data_client,
438  data_store_t & data_store,
439  size_t version,
440  Predicate && predicate,
441  bool sorted
442  )
443  {
444 
445  std::vector< accessor_t<T> > as;
446 
447  // the runtime id
448  auto runtime_id = data_client.runtime_id();
449 
450  // loop over each key pair
451  for (auto & entry_pair : data_store[NS]) {
452  // get the meta data key and label
453  const auto & meta_data_key = entry_pair.first;
454  auto & meta_data = entry_pair.second;
455  // now build the hash for this label
456  const auto & label = meta_data.label;
457  auto key_hash =
458  utils::hash<utils::const_string_t::hash_type_t>(label, label.size());
459  auto hash = key_hash ^ runtime_id;
460  // filter out the accessors for different data_clients
461  if ( meta_data_key != hash ) continue;
462  // if the reconstructed hash matches the meta data key,
463  // then we may want this one
464  auto a = get_accessor<T>( meta_data, version );
465  if ( a )
466  if (meta_data.rtti->type_info == typeid(T) && predicate(a))
467  as.emplace_back( std::move(a) );
468  } // for
469 
470  // if sorting is requested
471  if (sorted)
472  std::sort(
473  as.begin(), as.end(),
474  [](const auto & a, const auto &b) { return a.label()<b.label(); }
475  );
476 
477  return as;
