90  entry_value_t * end = start + n;

93  start, end, entry_value_t(ragged_index),
94  [](const entry_value_t & e1, const entry_value_t & e2) -> bool {
95  return e1.entry < e2.entry;

101  return itr->value;

103 
104  while (end != itr) {
