24 /*
25 
26  Description of major features and terms specific to FleCSI:
27 
28  mesh dimension MD - e.g: MD = 2 for a 2d mesh. We currently support 2d and 3d
29  meshes.
30 
31  topological dimension D - the dimensionality associated with entities,
32  e.g: D = 0 is interpreted as a vertex, D = 1 an edge or face for MD = 2,
33  D = 2 is a cell for MD = 2
34 
35  domain M - a sub-mesh or mesh space that holds entities of various topological
36  dimension
37 
38  connectivity - a directed connection or adjancy between entities of differing
39  topological dimension in the same domain. e.g: D1 -> D2 (edges -> faces)
40  for MD = 3. Cell to vertex connectivity is supplied by the user and all
41  other connectivies are computed by the topology.
42 
43  binding - a type of connectivity that connects entities of potentially
44  differing topological dimension across two different domains
45 
46  entity - an object associated with a topological dimension, e.g: cell. Each
47  entity has an associated integer id.
48 
49  mesh topology - the top-level container for domains, entities,
50  and connectivities, referred to as the low-level interface
51 
52  mesh policy - the top-level class that a specialization creates to
53  parameterize the mesh topology to define such things as: mesh dimension,
54  number of domains, connectivity and binding pairs of interest, and entity
55  classes/types per each domain/topological dimension.
56 
57  entity set - contains an iterable set of entities. Support set operations such
58  as intersection, union, etc. and functional operations like apply, map,
59  reduce, etc. to apply a custom function to the set.
60 
61 */

606  template<typename I>
607  void compute_graph_partition(
608  size_t domain,
609  size_t dim,
610  const std::vector<I>& partition_sizes,
611  std::vector<mesh_graph_partition<I>>& partitions){
612 
613  using int_t = I;
614 
615  partitions.reserve(partition_sizes.size());
616 
617  int_t total_size = 0;
618  for(auto pi : partition_sizes){
619  total_size += pi;
620  }
621 
622  size_t n = num_entities_(dim, domain);
623  size_t pn = n / total_size;
624 
625  size_t to_dim;
626 
627  if (dim == 0) {
628  // vertex -> vertex via shared edge.
629  to_dim = 1;
630  } else {
631  // edge -> edge via shared vertex, cell -> cell via shared edge/face etc.
632  to_dim = dim - 1;
633  }
634 
635  const connectivity_t& c1 = get_connectivity(domain, dim, to_dim);
636  assert(!c1.empty() && "empty connectivity c1");
637  const index_vector_t& fv1 = c1.get_from_index_vec();
638 
639  const connectivity_t& c2 = get_connectivity(domain, to_dim, dim);
640  assert(!c2.empty() && "empty connectivity c2");
641  const index_vector_t& fv2 = c2.get_from_index_vec();
642 
643  mesh_graph_partition<int_t> cp;
644  cp.offset.reserve(pn);
645 
646  size_t offset = 0;
647  size_t pi = 0;
648 
649  std::vector<int_t> partition;
650  partition.push_back(0);
651 
652  for(size_t from_id = 0; from_id < n; ++from_id){
653  auto to_ids = c1.get_index_space().ids(fv1[from_id], fv1[from_id + 1]);
654  cp.offset.push_back(offset);
655 
656  for(auto to_id : to_ids){
657  auto ret_ids =
658  c2.get_index_space().ids(fv2[to_id.entity()], fv2[to_id.entity() + 1]);
659 
660  for(auto ret_id : ret_ids){
661  if(ret_id.entity() != from_id){
662  cp.index.push_back(ret_id.local_id());
663  ++offset;
664  }
665  }
666  }
667 
668  size_t m = cp.offset.size();
669 
670  if(m >= pn * partition_sizes[pi]){
671  partitions.emplace_back(std::move(cp));
672  partition.push_back(m + partition.back());
673  offset = 0;
674  ++pi;
675  }
676  }
677 
678  for(auto& pi : partitions){
679  pi.partition = partition;
680  }
681  }

921  typename std::enable_if< (UsingDimension > 1 &&
922  UsingDimension <= MT::num_dimensions) >::type
923  build_connectivity()
924  {
925  // std::cerr << "build: " << DimensionToBuild << " using " << UsingDimension << std::endl;
926 
927  // Sanity check
928  static_assert(
929  DimensionToBuild <= MT::num_dimensions,
930  "DimensionToBuild must be <= total number of dimensions"
931  );
932  static_assert(
933  UsingDimension <= MT::num_dimensions,
934  "UsingDimension must be <= total number of dimensions"
935  );
936  static_assert(
937  Domain < MT::num_domains,
938  "Domain must be < total number of domains"
939  );
940 
941  // Reference to storage from cells to the entity (to be created here).
942  connectivity_t & cell_to_entity =
943  get_connectivity_(Domain, UsingDimension, DimensionToBuild);
944 
945  // Storage for entity-to-vertex connectivity information.
946  connection_vector_t entity_vertex_conn;
947 
948  // Helper variables
949  size_t entity_id = 0;
950  size_t max_cell_entity_conns = 1;
951 
952  domain_connectivity<MT::num_dimensions> & dc = ms_.topology[Domain][Domain];
953 
954  // Get connectivity for cells to vertices.
955  connectivity_t & cell_to_vertex = dc.template get<UsingDimension>(0);
956  assert(!cell_to_vertex.empty());
957 
958  const size_t _num_cells = num_entities<UsingDimension, Domain>();
959 
960  // Storage for cell-to-entity connectivity information.
961  connection_vector_t cell_entity_conn(_num_cells);
962 
963  // This map is primarily used to make sure that entities are not
964  // created multiple times, i.e., that they are unique. The
965  // emplace method of the map is used to only define a new entity
966  // if it does not already exist in the map.
967  id_vector_map_t entity_vertices_map;
968 
969  // This buffer should be large enough to hold all entities
970  // vertices that potentially need to be created
971  std::array<id_t, 4096> entity_vertices;
972 
973  using cell_type = entity_type<UsingDimension, Domain>;
974  using entity_type = entity_type<DimensionToBuild, Domain>;
975 
976  auto& is = ms_.index_spaces[Domain][DimensionToBuild].template cast<
977  domain_entity<Domain, entity_type>>();
978  auto& cis = ms_.index_spaces[Domain][UsingDimension];
979 
980  for (size_t c = 0; c < _num_cells; ++c) {
981  // Get the cell object
982 
983  auto cell = static_cast<cell_type*>(cis[c]);
984 
985  id_t cell_id = cell->template global_id<Domain>();
986 
987  // Get storage reference.
988  id_vector_t & conns = cell_entity_conn[c];
989 
990  // Try to optimize storage.
991  conns.reserve(max_cell_entity_conns);
992 
993  // This call allows the users specialization to create
994  // whatever entities are needed to complete the mesh.
995  //
996  // p.first: The number of entities per cell.
997  // p.second: A std::vector of id_t containing the ids of the
998  // vertices that define the entity.
999  auto sv = cell->template create_entities(cell_id,
1000  DimensionToBuild, dc, entity_vertices.data());
1001 
1002  size_t n = sv.size();
1003 
1004  // iterate over the newly-defined entities
1005  for (size_t i = 0; i < n; ++i) {
1006  size_t m = sv[i];
1007 
1008  // Get the vertices that define this entity by getting
1009  // a pointer to the vector-of-vector data and then constructing
1010  // a vector of ids for only this entity.
1011  id_t * a = &entity_vertices[i * m];
1012  id_vector_t ev(a, a + m);
1013 
1014  // Sort the ids for the current entity so that they are
1015  // monotonically increasing. This ensures that entities are
1016  // created uniquely (using emplace_back below) because the ids
1017  // will always occur in the same order for the same entity.
1018  std::sort(ev.begin(), ev.end());
1019 
1020  // Emplace the sorted vertices into the entity map
1021  auto itr = entity_vertices_map.emplace(
1022  std::move(ev), id_t::make<DimensionToBuild, Domain>(
1023  entity_id, cell_id.partition()));
1024 
1025  // Add this id to the cell to entity connections
1026  conns.push_back(itr.first->second);
1027 
1028  // If the insertion took place
1029  if (itr.second) {
1030  // what does this do?
1031  id_vector_t ev2 = id_vector_t(a, a + m);
1032  entity_vertex_conn.emplace_back(std::move(ev2));
1033 
1034  max_cell_entity_conns =
1035  std::max(max_cell_entity_conns, conns.size());
1036 
1037  id_t global_id = id_t::make<Domain>(DimensionToBuild, entity_id);
1038 
1039  auto ent =
1040  MT::template create_entity<Domain, DimensionToBuild>(this, m);
1041  ent->template set_global_id<Domain>( global_id );
1042 
1043  is.push_back(static_cast<entity_type*>(ent));
1044 
1045  // A new entity was added, so we advance the id counter.
1046  ++entity_id;
1047  } // if
1048  } // for
1049  } // for
1050 
1051  // Set the connectivity information from the created entities to
1052  // the vertices.
1053  connectivity_t & entity_to_vertex = dc.template get<DimensionToBuild>(0);
1054  entity_to_vertex.init(entity_vertex_conn);
1055  cell_to_entity.init(cell_entity_conn);
1056  } // build_connectivity

1100  template <size_t FM, size_t TM, size_t FD, size_t TD, size_t D>
1101  void intersect()
1102  {
1103  // std::cerr << "intersect: " << FD << " -> " << TD << std::endl;
1104 
1105  // The connectivity we will be populating
1106  connectivity_t & out_conn = get_connectivity_(FM, TM, FD, TD);
1107  if (!out_conn.empty()) {
1108  return;
1109  } // if
1110 
1111  // the number of each entity type
1112  auto num_from_ent = num_entities_(FD, FM);
1113  auto num_to_ent = num_entities_(TD, FM);
1114 
1115  // Temporary storage for connection id's
1116  connection_vector_t conns(num_from_ent);
1117 
1118  // Keep track of which to id's we have visited
1119  using visited_vec = std::vector<bool>;
1120  visited_vec visited(num_to_ent);
1121 
1122  size_t max_size = 1;
1123 
1124  // Read connectivities
1125  connectivity_t & c = get_connectivity_(FM, FD, D);
1126  assert(!c.empty());
1127 
1128  connectivity_t & c2 = get_connectivity_(TM, TD, D);
1129  assert(!c2.empty());
1130 
1131  // Iterate through entities in from topological dimension
1132  for (auto from_entity : entities<FD, FM>()) {
1133  id_t from_id = from_entity->template global_id<FM>();
1134  id_vector_t & ents = conns[from_id.entity()];
1135  ents.reserve(max_size);
1136 
1137  size_t count;
1138  id_t * ep = c.get_entities(from_id.entity(), count);
1139 
1140  // Create a copy of to vertices so they can be sorted
1141  id_vector_t from_verts(ep, ep+count);
1142  // sort so we have a unique key for from vertices
1143  std::sort(from_verts.begin(), from_verts.end());
1144 
1145  // initially set all to id's to unvisited
1146  for (auto from_ent2 : entities<D, FM>(from_entity)) {
1147  for (id_t to_id : entity_ids<TD, TM>(from_ent2)) {
1148  visited[to_id.entity()] = false;
1149  }
1150  }
1151 
1152  // Loop through each from entity again
1153  for (auto from_ent2 : entities<D, FM>(from_entity)) {
1154  for (id_t to_id : entity_ids<TD, TM>(from_ent2)) {
1155 
1156  // If we have already visited, skip
1157  if (visited[to_id.entity()]) {
1158  continue;
1159  } // if
1160 
1161  visited[to_id.entity()] = true;
1162 
1163  // If the topological dimensions are the same, always add to id
1164  if (FD == TD) {
1165  if (from_id != to_id) {
1166  ents.push_back(to_id);
1167  } // if
1168  } else {
1169  size_t count;
1170  id_t * ep = c2.get_entities(to_id.entity(), count);
1171 
1172  // Create a copy of to vertices so they can be sorted
1173  id_vector_t to_verts(ep, ep + count);
1174  // Sort to verts so we can do an inclusion check
1175  std::sort(to_verts.begin(), to_verts.end());
1176 
1177  // If from vertices contains the to vertices add to id
1178  // to this connection set
1179  if ( D < TD ) {
1180  if ( std::includes( from_verts.begin(), from_verts.end(),
1181  to_verts.begin(), to_verts.end()) )
1182  ents.emplace_back(to_id);
1183  }
1184  // If we are going through a higher level, then set
1185  // intersection is sufficient. i.e. one set does not need to
1186  // be a subset of the other
1187  else {
1188  if ( utils::intersects( from_verts.begin(), from_verts.end(),
1189  to_verts.begin(), to_verts.end() ) )
1190  ents.emplace_back(to_id);
1191  } // if
1192 
1193  } // if
1194  } // for
1195  } // for
1196 
1197  max_size = std::max(ents.size(), max_size);
1198  } // for
1199 
1200  // Finally create the connection from the temporary conns
1201  out_conn.init(conns);
1202  } // intersect

1208  template <size_t M, size_t FD, size_t TD>
1209  void compute_connectivity()
1210  {
1211  // std::cerr << "compute: " << FD << " -> " << TD << std::endl;
1212 
1213  // Get the output connectivity
1214  connectivity_t & out_conn = get_connectivity_(M, FD, TD);
1215 
1216  // Check if we have already computed it
1217  if (!out_conn.empty()) {
1218  return;
1219  } // if
1220 
1221  // if we don't have cell -> vertex connectivities, then
1222  // try building cell -> vertex connectivity through the
1223  // faces (3d) or edges(2d)
1224  static_assert( MT::num_dimensions <= 3,
1225  "this needs to be re-thought for higher dimensions" );
1226 
1227  if ( get_connectivity_(M, MT::num_dimensions, 0).empty() ) {
1228  assert( !get_connectivity_(M, MT::num_dimensions-1, 0).empty() &&
1229  " need at least edges(2d)/faces(3) -> vertex connectivity" );
1230  // assume we have cell -> faces, so invert it to get faces -> cells
1231  transpose<M, M, MT::num_dimensions-1, MT::num_dimensions>();
1232  // invert faces -> vertices to get vertices -> faces
1233  transpose<M, M, 0, MT::num_dimensions-1>();
1234  // build cells -> vertices via intersections with faces
1235  intersect<M, M, MT::num_dimensions, 0, MT::num_dimensions-1>();
1236  }
1237 
1238  // Check if we need to build entities, e.g: edges or faces
1239  if (num_entities_(FD, M) == 0) {
1240  if ( get_connectivity_(M, FD+1, 0).empty() )
1241  build_connectivity<M, FD, MT::num_dimensions>();
1242  else
1243  build_connectivity<M, FD, FD+1>();
1244  } // if
1245 
1246  if (num_entities_(TD, M) == 0) {
1247  if ( get_connectivity_(M, TD+1, 0).empty() )
1248  build_connectivity<M, TD, MT::num_dimensions>();
1249  else
1250  build_connectivity<M, TD, TD+1>();
1251  } // if
1252 
1253  if (num_entities_(FD, M) == 0 && num_entities_(TD, M) == 0) {
1254  return;
1255  } // if
1256 
1257  // Depending on the corresponding topological dimensions, call transpose
1258  // or intersect as need
1259  if (FD < TD) {
1260  compute_connectivity<M, TD, FD>();
1261  transpose<M, M, FD, TD>();
1262  } else {
1263  if (FD == 0 && TD == 0) {
1264  // compute vertex to vertex connectivities through shared cells.
1265  compute_connectivity<M, FD, MT::num_dimensions>();
1266  compute_connectivity<M, MT::num_dimensions, TD>();
1267  intersect<M, M, FD, TD, MT::num_dimensions>();
1268  } else {
1269  // computer connectivities through shared vertices.
1270  compute_connectivity<M, FD, 0>();
1271  compute_connectivity<M, 0, TD>();
1272  intersect<M, M, FD, TD, 0>();
1273  }
1274  } // if
1275  } // compute_connectivity
