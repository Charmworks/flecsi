24 /*
25 
26  Description of major features and terms specific to FleCSI:
27 
28  mesh dimension MD - e.g: MD = 2 for a 2d mesh. We currently support 2d and 3d
29  meshes.
30 
31  topological dimension D - the dimensionality associated with entities,
32  e.g: D = 0 is interpreted as a vertex, D = 1 an edge or face for MD = 2,
33  D = 2 is a cell for MD = 2
34 
35  domain M - a sub-mesh or mesh space that holds entities of various topological
36  dimension
37 
38  connectivity - a directed connection or adjancy between entities of differing
39  topological dimension in the same domain. e.g: D1 -> D2 (edges -> faces)
40  for MD = 3. Cell to vertex connectivity is supplied by the user and all
41  other connectivies are computed by the topology.
42 
43  binding - a type of connectivity that connects entities of potentially
44  differing topological dimension across two different domains
45 
46  entity - an object associated with a topological dimension, e.g: cell. Each
47  entity has an associated integer id.
48 
49  mesh topology - the top-level container for domains, entities,
50  and connectivities, referred to as the low-level interface
51 
52  mesh policy - the top-level class that a specialization creates to
53  parameterize the mesh topology to define such things as: mesh dimension,
54  number of domains, connectivity and binding pairs of interest, and entity
55  classes/types per each domain/topological dimension.
56 
57  entity set - contains an iterable set of entities. Support set operations such
58  as intersection, union, etc. and functional operations like apply, map,
59  reduce, etc. to apply a custom function to the set.
60 
61 */

1104  template<typename I>
1105  void compute_graph_partition(
1106  size_t domain,
1107  size_t dim,
1108  const std::vector<I>& partition_sizes,
1109  std::vector<mesh_graph_partition<I>>& partitions){
1110 
1111  using int_t = I;
1112 
1113  partitions.reserve(partition_sizes.size());
1114 
1115  int_t total_size = 0;
1116  for(auto pi : partition_sizes){
1117  total_size += pi;
1118  }
1119 
1120  size_t n = num_entities_(dim, domain);
1121  size_t pn = n / total_size;
1122 
1123  size_t to_dim;
1124 
1125  if (dim == 0) {
1126  // vertex -> vertex via shared edge.
1127  to_dim = 1;
1128  } else {
1129  // edge -> edge via shared vertex, cell -> cell via shared edge/face etc.
1130  to_dim = dim - 1;
1131  }
1132 
1133  const connectivity_t& c1 = get_connectivity(domain, dim, to_dim);
1134  assert(!c1.empty() && "empty connectivity c1");
1135  const index_vector_t& fv1 = c1.get_from_index_vec();
1136 
1137  const connectivity_t& c2 = get_connectivity(domain, to_dim, dim);
1138  assert(!c2.empty() && "empty connectivity c2");
1139  const index_vector_t& fv2 = c2.get_from_index_vec();
1140 
1141  mesh_graph_partition<int_t> cp;
1142  cp.offset.reserve(pn);
1143 
1144  size_t offset = 0;
1145  size_t pi = 0;
1146 
1147  std::vector<int_t> partition;
1148  partition.push_back(0);
1149 
1150  for(size_t from_id = 0; from_id < n; ++from_id){
1151  auto to_ids = id_range(c1.get_entities(), fv1[from_id], fv1[from_id + 1]);
1152  cp.offset.push_back(offset);
1153 
1154  for(auto to_id : to_ids){
1155  auto ret_ids = id_range(c2.get_entities(), fv2[to_id.entity()], fv2[to_id.entity() + 1]);
1156 
1157  for(auto ret_id : ret_ids){
1158  if(ret_id.entity() != from_id){
1159  cp.index.push_back(ret_id.local_id());
1160  ++offset;
1161  }
1162  }
1163  }
1164 
1165  size_t m = cp.offset.size();
1166 
1167  if(m >= pn * partition_sizes[pi]){
1168  partitions.emplace_back(std::move(cp));
1169  partition.push_back(m + partition.back());
1170  offset = 0;
1171  ++pi;
1172  }
1173  }
1174 
1175  for(auto& pi : partitions){
1176  pi.partition = partition;
1177  }
1178  }

1388  typename std::enable_if< (UsingDimension > 1 && UsingDimension <= MT::num_dimensions) >::type
1389  build_connectivity()

1392

1394  static_assert(
1395  DimensionToBuild <= MT::num_dimensions,
1396  "DimensionToBuild must be <= total number of dimensions"
1397  );
1398  static_assert(
1399  UsingDimension <= MT::num_dimensions,
1400  "UsingDimension must be <= total number of dimensions"
1401  );
1402  static_assert(
1403  Domain < MT::num_domains,
1404  "Domain must be < total number of domains"
1405  );
1406 
1407  // Reference to storage from cells to the entity (to be created here).
1408  connectivity_t & cell_to_entity = get_connectivity_(Domain, UsingDimension, DimensionToBuild);
1409 
1410  // Storage for entity-to-vertex connectivity information.
1411  connection_vector_t entity_vertex_conn;
1412 
1413  // Helper variables
1414  size_t entity_id = 0;
1415  size_t max_cell_entity_conns = 1;
1416 
1417  domain_connectivity<MT::num_dimensions> & dc = ms_.topology[Domain][Domain];
1418 
1419  // Get connectivity for cells to vertices.
1420  connectivity_t & cell_to_vertex = dc.template get<UsingDimension>(0);
1421  assert(!cell_to_vertex.empty());
1422 
1423  const size_t _num_cells = num_entities<UsingDimension, Domain>();
1424 
1425  // Storage for cell-to-entity connectivity information.
1426  connection_vector_t cell_entity_conn(_num_cells);
1427 
1428  // This map is primarily used to make sure that entities are not
1429  // created multiple times, i.e., that they are unique. The
1430  // emplace method of the map is used to only define a new entity
1431  // if it does not already exist in the map.
1432  id_vector_map_t entity_vertices_map;
1433 
1434  // This buffer should be large enough to hold all entities
1435  // vertices that potentially need to be created
1436  std::array<id_t, 4096> entity_vertices;
1437 
1438  for (size_t c = 0; c < _num_cells; ++c) {
1439  // Get the cell object
1440  auto cell = static_cast<entity_type<UsingDimension, Domain> *>(
1441  ms_.entities[Domain][UsingDimension][c]);
1442 
1443  id_t cell_id = cell->template global_id<Domain>();
1444 
1445  // Get storage reference.
1446  id_vector_t & conns = cell_entity_conn[c];
1447 
1448  // Try to optimize storage.
1449  conns.reserve(max_cell_entity_conns);
1450 
1451  // Get the vertices that define the current cell
1452  size_t end_index;
1453  id_t * vertices = cell_to_vertex.get_entities(c, end_index);
1454 
1455  // This call allows the users specialization to create
1456  // whatever entities are needed to complete the mesh.
1457  //
1458  // p.first: The number of entities per cell.
1459  // p.second: A std::vector of id_t containing the ids of the
1460  // vertices that define the entity.
1461  auto sv =
1462  cell->template create_entities(cell_id, DimensionToBuild, dc, entity_vertices.data());
1463 
1464  size_t n = sv.size();
1465 
1466  // iterate over the newly-defined entities
1467  for (size_t i = 0; i < n; ++i) {
1468  size_t m = sv[i];
1469 
1470  // Get the vertices that define this entity by getting
1471  // a pointer to the vector-of-vector data and then constructing
1472  // a vector of ids for only this entity.
1473  id_t * a = &entity_vertices[i * m];
1474  id_vector_t ev(a, a + m);
1475 
1476  // Sort the ids for the current entity so that they are
1477  // monotonically increasing. This ensures that entities are
1478  // created uniquely (using emplace_back below) because the ids
1479  // will always occur in the same order for the same entity.
1480  std::sort(ev.begin(), ev.end());
1481 
1482  // Emplace the sorted vertices into the entity map
1483  auto itr = entity_vertices_map.emplace(
1484  std::move(ev), id_t::make<DimensionToBuild, Domain>(entity_id, cell_id.partition()));
