209 
210  // size of floating point variables used in app.
211  int CPU_word_size = sizeof(real_t);
212  // size of floating point to be stored in file.
213  int IO_word_size = sizeof(real_t);
214  auto exoid =
215  ex_create(name.c_str(), EX_CLOBBER, &CPU_word_size, &IO_word_size);
216  assert(exoid >= 0);
217  auto d = m.dimension();

219  auto num_elem = m.num_cells();

221  auto num_node_sets = 0;
222  auto num_side_sets = 0;
223 
224  // initialize the file.
225  auto status = ex_put_init(exoid, "Exodus II output from flecsi.", d,
226  num_nodes, num_elem, num_elem_blk, num_node_sets, num_side_sets);
227  assert(status == 0);
228 
229  // get the coordinates from the mesh.
230  real_t * xcoord = new real_t[num_nodes];
231  real_t * ycoord = new real_t[num_nodes];
232  auto i = 0;
233  for (auto v : m.vertices()) {
234  xcoord[i] = v->coordinates()[0];
235  ycoord[i] = v->coordinates()[1];
236  i++;
237  } // for
238  // write the coordinates to the file
239  status = ex_put_coord(exoid, xcoord, ycoord, nullptr);
240  assert(status == 0);
241 
242  delete[] xcoord;
243  delete[] ycoord;
244 
245  // write the coordinate names
246  const char * coord_names[3];
247  coord_names[0] = "x";
248  coord_names[1] = "y";
249  coord_names[2] = "z";
250  status = ex_put_coord_names(exoid, (char **)coord_names);
251  assert(status == 0);
252 
253  // loop over element blocks
254  auto blockid = 0;
255  auto num_attr = 0;
256  auto num_nodes_per_elem = 4;
257  status = ex_put_elem_block(
258  exoid, blockid, "quad", num_elem, num_nodes_per_elem, num_attr);
259  assert(status == 0);
260 
261  // element definitions
262  int * elt_conn = new int[num_elem * num_nodes_per_elem];
263  i = 0;
264  for (auto c : m.cells()) {
265  for (auto v : m.vertices(c)) {
266  elt_conn[i] = v.id() + 1; // 1 based index in exodus
267  i++;
268  } // for
269  } // for
270 
271  // write connectivity
272  status = ex_put_elem_conn(exoid, blockid, elt_conn);
273  assert(status == 0);
274 
275  delete[] elt_conn;
276 
277  // write field data
278 
279  // nodal field data
280  int num_nf = 0; // number of nodal fields
281  // real scalars persistent at vertices
282  auto rspav = access_type_if(m, real_t, is_persistent_at(vertices));
283  num_nf += rspav.size();
284  // int scalars persistent at vertices
285  auto ispav = access_type_if(m, int, is_persistent_at(vertices));
286  num_nf += ispav.size();
287  // real vectors persistent at vertices
288  auto rvpav = access_type_if(m, vector_t, is_persistent_at(vertices));
289  num_nf += m.dimension() * rvpav.size();
290 
291  // variable extension for vectors
292  std::string var_ext[3];
293  var_ext[0] = "_x";
294  var_ext[1] = "_y";
295  var_ext[2] = "_z";
296 
297  // node variable names array
298  char ** node_var_names = new char *[num_nf]();
299  for (int i = 0; i < num_nf; ++i) {
300  node_var_names[i] = new char[MAX_STR_LENGTH]();
301  memset(node_var_names[i], 0x00, MAX_STR_LENGTH);
302  } // for
303 
304  // fill node variable names array
305  int inum = 0;
306  for (auto sf : rspav) {
307  std::copy(sf.label().begin(), sf.label().end(), node_var_names[inum]);
308  inum++;
309  } // for
310  for (auto sf : ispav) {
311  std::copy(sf.label().begin(), sf.label().end(), node_var_names[inum]);
312  inum++;
313  } // for
314  for (auto vf : rvpav) {
315  size_t len = vf.label().size();
316  for (size_t d = 0; d < m.dimension(); ++d) {
317  std::copy(vf.label().begin(), vf.label().end(), node_var_names[inum]);
318  std::copy(
319  var_ext[d].begin(), var_ext[d].end(), &node_var_names[inum][len]);
320  inum++;
321  } // for
322  } // for
323 
324  // put the number of nodal fields
325  if (num_nf > 0) {
326  status = ex_put_var_param(exoid, "n", num_nf);
327  assert(status == 0);
328  status = ex_put_var_names(exoid, "n", num_nf, node_var_names);
329  assert(status == 0);
330  } // if
331 
332  // write nodal fields
333  inum = 1;
334  // node field buffer
335  real_t * nf = new real_t[num_nodes];
336  for (auto sf : rspav) {
337  for (auto v : m.vertices())
338  nf[v.id()] = sf[v];
339  status = ex_put_nodal_var(exoid, 1, inum, num_nodes, nf);
340  assert(status == 0);
341  inum++;
342  } // for
343  for (auto sf : ispav) {
344  // cast int fields to real_t
345  for (auto v : m.vertices())
346  nf[v.id()] = (real_t)sf[v];
347  status = ex_put_nodal_var(exoid, 1, inum, num_nodes, nf);
348  assert(status == 0);
349  inum++;
350  } // for
351  for (auto vf : rvpav) {
352  for (size_t d = 0; d < m.dimension(); ++d) {
353  for (auto v : m.vertices())
354  nf[v.id()] = vf[v][d];
355  status = ex_put_nodal_var(exoid, 1, inum, num_nodes, nf);
356  assert(status == 0);
357  inum++;
358  } // for
359  } // for
360 
361  // clean up
362  for (int i = 0; i < num_nf; ++i) {
363  delete[] node_var_names[i];
364  } // for
365 
366  delete[] node_var_names;
367  delete[] nf;
368 
369  // element field data
370  int num_ef = 0; // number of element fields
371  // real scalars persistent at cells
372  auto rspac = access_type_if(m, real_t, is_persistent_at(cells));
373  num_ef += rspac.size();
374  // int scalars persistent at cells
375  auto ispac = access_type_if(m, int, is_persistent_at(cells));
376  num_ef += ispac.size();
377  // real vectors persistent at cells
378  auto rvpac = access_type_if(m, vector_t, is_persistent_at(cells));
379  num_ef += m.dimension() * rvpac.size();
380 
381  // element variable names array
382  char ** elem_var_names = new char *[num_ef]();
383  for (int i = 0; i < num_ef; ++i) {
384  elem_var_names[i] = new char[MAX_STR_LENGTH];
385  memset(elem_var_names[i], 0x00, MAX_STR_LENGTH);
386  } // for
387 
388  // fill element variable names array
389  inum = 0;
390  for (auto sf : rspac) {
391  std::copy(sf.label().begin(), sf.label().end(), elem_var_names[inum]);
392  inum++;
393  } // for
394  for (auto sf : ispac) {
395  std::copy(sf.label().begin(), sf.label().end(), elem_var_names[inum]);
396  inum++;
397  } // for
398  for (auto vf : rvpac) {
399  size_t len = vf.label().size();
400  for (size_t d = 0; d < m.dimension(); ++d) {
401  std::copy(vf.label().begin(), vf.label().end(), elem_var_names[inum]);
402  std::copy(
403  var_ext[d].begin(), var_ext[d].end(), &elem_var_names[inum][len]);
404  inum++;
405  } // for
406  } // for
407 
408  // put the number of element fields
409  if (num_ef > 0) {
410  status = ex_put_var_param(exoid, "e", num_ef);
411  assert(status == 0);
412  status = ex_put_var_names(exoid, "e", num_ef, elem_var_names);
413  assert(status == 0);
414  } // if
415 
416  // write element fields
417  inum = 1;
418  // element field buffer
419  real_t * ef = new real_t[num_elem];
420  for (auto sf : rspac) {
421  for (auto c : m.cells())
422  ef[c.id()] = sf[c];
423  status = ex_put_elem_var(exoid, 1, inum, 0, num_elem, ef);
424  assert(status == 0);
425  inum++;
426  } // for
427  for (auto sf : ispac) {
428  // cast int fields to real_t
429  for (auto c : m.cells())
430  ef[c.id()] = (real_t)sf[c];
431  status = ex_put_elem_var(exoid, 1, inum, 0, num_elem, ef);
432  assert(status == 0);
433  inum++;
434  } // for
435  for (auto vf : rvpac) {
436  for (size_t d = 0; d < m.dimension(); ++d) {
437  for (auto c : m.cells())
438  ef[c.id()] = vf[c][d];
439  status = ex_put_elem_var(exoid, 1, inum, 0, num_elem, ef);
440  assert(status == 0);
441  inum++;
442  } // for
443  } // for

447  delete[] elem_var_names[i];
448  } // for
