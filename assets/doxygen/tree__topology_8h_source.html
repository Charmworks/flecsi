38  Tree topology is a statically configured N-dimensional hashed tree for
39  representing localized entities, e.g. particles. It stores entities in a
40  configurable branch type. Inserting entities into a branch can cause that
41  branch to be refined or coarsened correspondingly. A client of tree topology

43  compile-time parameters. Specializations can define a policy and default
44  branch types which can then be specialized in a simpler fashion
45  (see the basic_tree specialization).

321 public:

332  //-----------------------------------------------------------------//
336  //-----------------------------------------------------------------//
337  template<typename S>
338  branch_id_u(
339  const std::array<point<S, dimension>, 2> & range,
340  const point<S, dimension> & p,
341  size_t depth)
342  : id_(int_t(1) << depth * dimension + (bits - 1) % dimension) {
343  std::array<int_t, dimension> coords;
344 
345  for (size_t i = 0; i < dimension; ++i) {
346  S min = range[0][i];
347  S scale = range[1][i] - min;
348  coords[i] = (p[i] - min) / scale * (int_t(1) << (bits - 1) / dimension);
349  }
350 
351  size_t k = 0;
352  for (size_t i = max_depth - depth; i < max_depth; ++i) {
353  for (size_t j = 0; j < dimension; ++j) {
354  int_t bit = (coords[j] & int_t(1) << i) >> i;
355  id_ |= bit << (k * dimension + j);
356  }
357  ++k;
358  }
359  }
