410  void build_reverse_intermediate_maps(bool reset = false, bool sort = false) {
411 
412  // clear the map for safety
413  if (reset)
414  reverse_intermediate_map_.clear();
415 
416  // now flip all the mappings
417  for (const auto & forward_map : intermediate_map_) {
418  auto key = forward_map.first;
419  auto & reverse_map = reverse_intermediate_map_[key];
420  // it will be empty if never set, or reset==true
421  if (reverse_map.empty()) {
422 
423  // assume unsorted
424  if (sort) {
425  for (auto & entry : forward_map.second) {
426  std::sort(entry.second.begin(), entry.second.end());
427  reverse_map[entry.second] = entry.first;
428  }
429  }
430  // assume sorted
431  else {
432  for (const auto & entry : forward_map.second)
433  reverse_map[entry.second] = entry.first;
434  }
435  }
436  }
437  }
