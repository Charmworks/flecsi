32 template <unsigned DIM, typename T>
33 inline bool
34 offsetsAreDense(const LegionRuntime::Arrays::Rect<DIM> &bounds,
35  const LegionRuntime::Accessor::ByteOffset *offset)
36 {
37  off_t exp_offset = sizeof(T);
38  for (unsigned i = 0; i < DIM; i++) {
39  bool found = false;
40  for (unsigned j = 0; j < DIM; j++)
41  if (offset[j].offset == exp_offset) {
42  found = true;
43  exp_offset *= (bounds.hi[j] - bounds.lo[j] + 1);
44  break;
45  }
46  if (!found) return false;
47  }
48  return true;
49 }

261  void
262  partition(
263  uint64_t nParts,
264  LegionRuntime::HighLevel::Context ctx,
265  LegionRuntime::HighLevel::HighLevelRuntime *lrt
266  ) {
267  using LegionRuntime::Arrays::Rect;
268 
269  // For now only allow even partitioning.
270  assert(0 == length % nParts && "Uneven partitioning requested.");
271  //
272  uint64_t inc = length / nParts; // the increment
273  Rect<1> colorBounds(Point<1>(0), Point<1>(nParts - 1));
274  LegionRuntime::HighLevel::Domain colorDomain =
275  LegionRuntime::HighLevel:: Domain::from_rect<1>(colorBounds);
276  // +
277  // |
278  // |
279  // (x1)-+-+
280  // | |
281  // | m / nSubregions
282  // (x0) + |
283  uint64_t x0 = 0, x1 = inc - 1;
284  LegionRuntime::HighLevel::DomainColoring disjointColoring;
285  // a list of sub-grid bounds.
286  // provides a task ID to sub-grid bounds mapping.
287  std::vector< Rect<1> > subgridBnds;
288  for (uint64_t color = 0; color < nParts; ++color) {
289  Rect<1> subRect((Point<1>(x0)), (Point<1>(x1)));
290  // cache the subgrid bounds
291  subgridBnds.push_back(subRect);
292 #if 0 // nice debug
293  printf("vec disjoint partition: (%d) to (%d)\n",
294  subRect.lo.x[0], subRect.hi.x[0]);
295 #endif
296  disjointColoring[color] =
297  LegionRuntime::HighLevel::Domain::from_rect<1>(subRect);
298  x0 += inc;
299  x1 += inc;
300  }
301  auto iPart = lrt->create_index_partition(
302  ctx, mIndexSpace,
303  colorDomain, disjointColoring,
304  true /* disjoint */
305  );
306  // logical partitions
307  auto lp = lrt->get_logical_partition(ctx, logicalRegion, iPart);
308  // launch domain -- one task per color
309  // launch domain
310  LegionRuntime::HighLevel::Domain lDom = colorDomain;
311  // add to the vector of partitions
312  mPVec.push_back(PVecItem(subgridBnds, lDom, lp));
313  }
