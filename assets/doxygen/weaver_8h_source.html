42  weaver(flecsi::topology::graph_definition_t & gd)
43  :
44  gd_(gd)
45  {
46  int size;
47  int rank;
48 
49  MPI_Comm_size(MPI_COMM_WORLD, &size);
50  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
51 
52  // Create the dCRS representation for the distributed
53  // partitioner.
54  auto dcrs = flecsi::dmp::make_dcrs(gd_);
55 
56  // Create a partitioner instance to generate the primary partition.
57  auto partitioner = std::make_shared<flecsi::dmp::parmetis_partitioner_t>();
58 
59  // Create the primary partition.
60  //auto primary = partitioner->partition(dcrs);
61  primary_cells = partitioner->partition(dcrs);
62 
63  // Compute the dependency closure of the primary cell partition
64  // through vertex intersections (specified by last argument "1").
65  // To specify edge or face intersections, use 2 (edges) or 3 (faces).
66  // FIXME: We may need to replace this with a predicate function.
67  auto closure = flecsi::topology::entity_closure<2,2,0>(gd_, primary_cells);
68 
69  // Subtracting out the initial set leaves just the nearest
70  // neighbors. This is similar to the image of the adjacency
71  // graph of the initial indices.
72  auto nearest_neighbors =
73  flecsi::utils::set_difference(closure, primary_cells);
74 
75  // The closure of the nearest neighbors intersected with
76  // the initial indeces gives the shared indices. This is similar to
77  // the preimage of the nearest neighbors.
78  auto nearest_neighbor_closure =
79  flecsi::topology::entity_closure<2,2,0>(gd_, nearest_neighbors);
80 
81  // We can iteratively add halos of nearest neighbors, e.g.,
82  // here we add the next nearest neighbors. For most mesh types
83  // we actually need information about the ownership of these indices
84  // so that we can deterministically assign rank ownership to vertices.
85  auto next_nearest_neighbors =
86  flecsi::utils::set_difference(nearest_neighbor_closure, closure);
87 
88  // The union of the nearest and next-nearest neighbors gives us all
89  // of the cells that might reference a vertex that we need.
90  auto all_neighbors = flecsi::utils::set_union(nearest_neighbors,
91  next_nearest_neighbors);
92 
93  // Create a communicator instance to get neighbor information.
94  auto communicator = std::make_shared<flecsi::dmp::mpi_communicator_t>();
95 
96  // Get the rank and offset information for our nearest neighbor
97  // dependencies. This also gives information about the ranks
98  // that access our shared cells.
99  auto cell_nn_info = communicator->get_cell_info(primary_cells,
100  nearest_neighbors);
101 
102  //
103  auto cell_all_info = communicator->get_cell_info(primary_cells,
104  all_neighbors);
105 
106  // TODO: seperate this part into its own function.
107  // Create a map version of the local info for lookups below.
108  std::unordered_map <size_t, size_t> primary_indices_map;
109  {
110  size_t offset(0);
111  for (auto i: primary_cells) {
112  primary_indices_map[offset++] = i;
113  } // for
114  } // scope
115 
116  // Create a map version of the remote info for lookups below.
117  std::unordered_map <size_t, entry_info_t> remote_info_map;
118  for (auto i: std::get<1>(cell_all_info)) {
119  remote_info_map[i.id] = i;
120  } // for
121 
122 
123  // Populate exclusive and shared cell information.
124  {
125  size_t offset(0);
126  for (auto i: std::get<0>(cell_nn_info)) {
127  if (i.size()) {
128  shared_cells.insert(entry_info_t(primary_indices_map[offset],
129  rank, offset, i));
130  } else {
131  exclusive_cells.insert(entry_info_t(primary_indices_map[offset],
132  rank, offset, i));
133  } // if
134  ++offset;
135  } // for
136  } // scope
137 
138  // Populate ghost cell information.
139  {
140  size_t offset(0);
141  for (auto i: std::get<1>(cell_nn_info)) {
142  ghost_cells.insert(i);
143  } // for
144  } // scope
145 
146 
147  // TODO: move this into is own function
148  // Create a map version for lookups below.
149  std::unordered_map <size_t, entry_info_t> shared_cells_map;
150  {
151  for (auto i: shared_cells) {
152  shared_cells_map[i.id] = i;
153  } // for
154  } // scope
155 
156  // Form the vertex closure
157  auto vertex_closure = flecsi::topology::vertex_closure<2>(gd_, closure);
158 
159  // Assign vertex ownership
160  std::vector <std::set<size_t>> vertex_requests(size);
161  std::set <entry_info_t> vertex_info;
162 
163  size_t offset(0);
164  for (auto i: vertex_closure) {
165 
166  // Get the set of cells that reference this vertex.
167  auto referencers = flecsi::topology::vertex_referencers<2>(gd_, i);
168  size_t min_rank(std::numeric_limits<size_t>::max());
169  std::set <size_t> shared_vertices;
170 
171  for (auto c: referencers) {
172 
173  // If the referencing cell isn't in the remote info map
174  // it is a local cell.
175  if (remote_info_map.find(c) != remote_info_map.end()) {
176  min_rank = std::min(min_rank, remote_info_map[c].rank);
177  shared_vertices.insert(remote_info_map[c].rank);
178  } else {
179  min_rank = std::min(min_rank, size_t(rank));
180  raw_cell_vertex_conns.emplace_back(std::make_pair(c, i));
181  // If the local cell is shared, we need to add all of
182  // the ranks that reference it.
183  if (shared_cells_map.find(c) != shared_cells_map.end()) {
184  shared_vertices.insert(shared_cells_map[c].shared.begin(),
185  shared_cells_map[c].shared.end());
186  } // if
187  } // if
188  } // for
189 
190  if (min_rank == rank) {
191  // This is a vertex that belongs to our rank.
192  //auto entry = entry_info_t(i, rank, offset, shared_vertices);
193  vertex_info.insert(entry_info_t(i, rank, offset++, shared_vertices));
194  } else {
195  // Add remote vertex to the request for offset information.
196  vertex_requests[min_rank].insert(i);
197  } // fi
198  } // for
199 
200  auto vertex_offset_info =
201  communicator->get_vertex_info(vertex_info, vertex_requests);
202 
203  for (auto i: vertex_info) {
204  if (i.shared.size()) {
205  shared_vertices.insert(i);
206  } else {
207  exclusive_vertices.insert(i);
208  } // if
209  } // for
210 
211  for (auto i: shared_vertices) {
212  primary_vertices.insert(i.id);
213  }
214  for (auto i: exclusive_vertices) {
215  primary_vertices.insert(i.id);
216  }
217 
218  {
219  size_t r(0);
220  for (auto i: vertex_requests) {
221 
222  auto offset(vertex_offset_info[r].begin());
223  for (auto s: i) {
224  ghost_vertices.insert(entry_info_t(s, r, *offset));
225  ++offset;
226  } // for
227 
228  ++r;
229  } // for
230  } // scope
231  }
