417  vector_t side_facet_normal() const
418  {
419  // Use the edge midpoint and the cell centroid to create the normal vector.
420  // Multiply normal by the sign of the dot between the normal and the vector
421  // from v to e to get "outward facing" normal.
422  auto c = cell()->centroid();
423  auto e = edge()->midpoint();
424  auto v = vertex()->coordinates();
425  auto nrml = normal(c,e);
426  return nrml*sgn(dot(nrml,point_to_vector(e-v)));
427  }

433  vector_t cell_facet_normal() const
434  {

440  auto c = cell()->centroid();

551  B / \ C

565  vector_t B(point_to_vector(edge2()->midpoint() - xv));
