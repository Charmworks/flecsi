428  vector_t side_facet_normal() const
429  {
430  // Use the edge midpoint and the cell centroid to create the normal vector.
431  // Multiply normal by the sign of the dot between the normal and the vector
432  // from v to e to get "outward facing" normal.
433  auto c = cell()->centroid();
434  auto e = edge()->midpoint();
