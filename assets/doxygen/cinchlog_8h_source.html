225  static constexpr size_t usec_bytes = sizeof(suseconds_t);

227  packet_t(const char * msg = nullptr) {
228  timeval stamp;
229  if(gettimeofday(&stamp, NULL)) {
230  std::cerr << "CLOG: call to gettimeofday failed!!! " <<
231  __FILE__ << __LINE__ << std::endl;
232  std::exit(1);
233  } // if
234 
235  strncpy(data_, reinterpret_cast<const char *>(&stamp.tv_sec), sec_bytes);
236  strncpy(data_+sec_bytes, reinterpret_cast<const char *>(&stamp.tv_usec),
237  usec_bytes);
238 
239  std::ostringstream oss;
240  oss << msg;
241 
242  strcpy(data_+sec_bytes+usec_bytes, oss.str().c_str());

255  } // message

418  int
419  overflow(
420  int c
421  )
422  {
423  if(c == EOF) {
424  return !EOF;
425  }
426  else {
427  // Get the size before we add the current character
428  const size_t tbsize = test_buffer_.size();
429 
430  // Buffer the output for now...
431  test_buffer_.append(1, char(c)); // takes char
432 
433  switch(tbsize) {
434 
435  case 0:
436  if(c == '\033') {
437  // This could be a color string, start buffering
438  return c;
439  }
440  else {
441  // No match, go ahead and write the character
442  return flush_buffer(all_buffers);
443  } // if
444 
445  case 1:
446  if(c == '[') {
447  // This still looks like a color string, keep buffering
448  return c;
449  }
450  else {
451  // This is some other kind of escape. Write the
452  // buffered output to all buffers.
453  return flush_buffer(all_buffers);
454  } // if
455 
456  case 2:
457  if(c == '0' || c == '1') {
458  // This still looks like a color string, keep buffering
459  return c;
460  }
461  else {
462  // This is some other kind of escape. Write the
463  // buffered output to all buffers.
464  return flush_buffer(all_buffers);
465  } // if
466 
467  case 3:
468  if(c == ';') {
469  // This still looks like a color string, keep buffering
470  return c;
471  }
472  else if(c == 'm') {
473  // This is a plain color termination. Write the
474  // buffered output to the color buffers.
475  return flush_buffer(color_buffers);
476  }
477  else {
478  // This is some other kind of escape. Write the
479  // buffered output to all buffers.
480  return flush_buffer(all_buffers);
481  } // if
482 
483  case 4:
484  if(c == '3') {
485  // This still looks like a color string, keep buffering
486  return c;
487  }
488  else {
489  // This is some other kind of escape. Write the
490  // buffered output to all buffers.
491  return flush_buffer(all_buffers);
492  } // if
493 
494  case 5:
495  if(isdigit(c) && (c - '0') < 8) {
496  // This still looks like a color string, keep buffering
497  return c;
498  }
499  else {
500  // This is some other kind of escape. Write the
501  // buffered output to all buffers.
502  return flush_buffer(all_buffers);
503  } // if
504 
505  case 6:
506  if(c == 'm') {
507  // This is a color string termination. Write the
508  // buffered output to the color buffers.
509  return flush_buffer(color_buffers);
510  }
511  else {

712  void
713  init(std::string active = "none")
714  {
715  #if defined(CLOG_DEBUG)
716  std::cerr << COLOR_LTGRAY << "CLOG: initializing runtime" <<
717  COLOR_PLAIN << std::endl;
718  #endif
719 
720  // Because active tags are specified at runtime, it is
721  // necessary to maintain a map of the compile-time registered
722  // tag names to the id that they get assigned after the clog_t
723  // initialization (register_tag). This map will be used to populate
724  // the tag_bitset_ for fast runtime comparisons of enabled tag groups.
725 
726  // Note: For the time being, the map uses actual strings rather than
727  // hashes. We should consider creating a const_string_t type for
728  // constexpr string creation.
729 
730  // Initialize everything to false. This is the default, i.e., "none".
731  tag_bitset_.reset();
732 
733  // The default group is always active (unscoped). To avoid
734  // output for this tag, make sure to scope all CLOG output.
735  tag_bitset_.set(0);
736 
737  if(active == "all") {
738  // Turn on all of the bits for "all".
739  tag_bitset_.flip();
740  }
741  else if(active != "none") {
742  // Turn on the bits for the selected groups.
743 
744  #if defined(CLOG_DEBUG)
745  std::cerr << COLOR_LTGRAY << "CLOG: active tags (" <<
746  active << ")" << COLOR_PLAIN << std::endl;
747  #endif
748 
749  std::istringstream is(active);
750  std::string tag;
751  while(std::getline(is, tag, ',')) {
752  if(tag_map_.find(tag) != tag_map_.end()) {
753  tag_bitset_.set(tag_map_[tag]);
754  }
755  else {
756  std::cerr << "CLOG WARNING: " << tag <<
757  " has not been registered. Ignoring this group..." << std::endl;
758  } // if
759  } // while
760  } // if

763

766  COLOR_PLAIN << std::endl;
