89  const std::set<size_t> & request_indices,
90  size_t max_request_indices,
91  int colors) {
92  // Pad the request indices with size_t max. We will then set
93  // the indices of the actual request. Each rank that receives
94  // the request will try to provide information about the
95  // non size_t max values in the request. The others will
96  // be ignored.
97  std::vector<size_t> input_indices(
98  colors * max_request_indices, std::numeric_limits<size_t>::max());

100 
101  for (size_t c(0); c < colors; ++c) {
102  size_t off(0);
103  const size_t coff = c * max_request_indices;
104 
105  for (auto s : request_indices) {
106  input_indices[coff + off++] = s;
107  } // for
108  } // for
109 
110  const auto mpi_size_t_type =
111  flecsi::coloring::mpi_typetraits_u<size_t>::type();
112

114  int result = MPI_Alltoall(
115  &input_indices[0], max_request_indices, mpi_size_t_type,
116  &info_indices[0], max_request_indices, mpi_size_t_type, MPI_COMM_WORLD);

134  get_primary_info(
135  const std::set<size_t> & primary,
136  const std::set<size_t> & request_indices) override {
137  auto colors = size();
138  auto color = rank();
139 
140  // Store the request in a vector for indexed access below.
141  std::vector<size_t> request_indices_vector(
142  request_indices.begin(), request_indices.end());
143 
144  const auto mpi_size_t_type =
145  flecsi::coloring::mpi_typetraits_u<size_t>::type();
146 
147  size_t max_request_indices = get_max_request_size(request_indices.size());
148 
149  // Pad the request indices with size_t max. We will then set
150  // the indices of the actual request. Each rank that receives
151  // the request will try to provide information about the
152  // non size_t max values in the request. The others will
153  // be ignored.
154  std::vector<size_t> input_indices(
155  colors * max_request_indices, std::numeric_limits<size_t>::max());
156  std::vector<size_t> info_indices(colors * max_request_indices);
157  info_indices =
158  get_info_indices(request_indices, max_request_indices, colors);
159 
160  // For now, we need two arrays for each all-to-all communication:
161  // One for rank ownership of the request indices, and one
162  // for the offsets. We could probably combine these. However,
163  // we would probably have to define a custom MPI type. It
164  // will only be worth the effort if this appraoch is slow.
165  // The input offsets do not need to be initialized because
166  // the information is available in the input_indices array.
167  std::vector<size_t> input_offsets(colors * max_request_indices);
168  std::vector<size_t> info_offsets(colors * max_request_indices);
169 
170  // Reset input indices to use to send back information
171  std::fill(
172  input_indices.begin(), input_indices.end(),
173  std::numeric_limits<size_t>::max());
174 
175  // For the primary coloring, provide rank and entity information
176  // on indices that are shared with other processes.
177  std::vector<std::set<size_t>> local(primary.size());
178 
179  // See if we can fill any requests...
180  for (size_t r(0); r < colors; ++r) {
181 
182  // Ignore our rank
183  if (r == color) {
184  continue;
185  } // if
186 
187  // These array slices are just for convenience.
188  size_t * info = &info_indices[r * max_request_indices];
189  size_t * offset = &input_offsets[r * max_request_indices];
190  size_t * input = &input_indices[r * max_request_indices];
191 
192  // See which requests we can fulfill.
193  for (size_t i(0); i < max_request_indices; ++i) {
194 
195  auto match = primary.find(info[i]);
196 
197  if (match != primary.end()) {
198  // This is a match, i.e., we own this entity, so we can
199  // set the rank (ownership) and offset.
200  input[i] = color;
201  offset[i] = std::distance(primary.begin(), match);
202 
203  // We also need to register that this index is shared
204  // with other ranks
205  local[offset[i]].insert(r);
206  } // if
207  } // for
208  } // for
209 
210  // Send the indices information back to all ranks.
211  int result = MPI_Alltoall(
212  &input_indices[0], max_request_indices, mpi_size_t_type,
213  &info_indices[0], max_request_indices, mpi_size_t_type, MPI_COMM_WORLD);
214 
215  // Send the offsets information back to all ranks.
216  result = MPI_Alltoall(
217  &input_offsets[0], max_request_indices, mpi_size_t_type,
218  &info_offsets[0], max_request_indices, mpi_size_t_type, MPI_COMM_WORLD);
219 
220  std::set<entity_info_t> remote;
221 
222  // Collect all of the information for the remote entities.
223  for (size_t r(0); r < colors; ++r) {
224  // Skip these (we already know them!)
225  if (r == color) {
226  continue;
227  } // if
228 
229  // Another slice for convenience.
230  size_t * ranks = &info_indices[r * max_request_indices];
231  size_t * offsets = &info_offsets[r * max_request_indices];
232 
233  for (size_t i(0); i < max_request_indices; ++i) {
234 
235  if (ranks[i] != std::numeric_limits<size_t>::max()) {
236  // If this is not size_t max, this rank answered our request
237  // and we can set the information.
238  remote.insert(entity_info_t(
239  request_indices_vector[i], ranks[i], offsets[i], {}));
240  } // if
241  } // for
242  } // for
243 
244  return std::make_pair(local, remote);
245  } // get_primary_info

393  std::vector<std::set<size_t>> get_entity_info(
394  const std::set<entity_info_t> & entity_info,
395  const std::vector<std::set<size_t>> & request_indices) override {
396  auto colors = size();
397  auto color = rank();
398 
399  // Collect the size of each rank request to send.
400  std::vector<size_t> send_cnts(colors, 0);
401  for (size_t r(0); r < colors; ++r) {
402  send_cnts[r] = request_indices[r].size();
403  } // for
404 
405  // Send the request size (in indices) to each rank.
406  std::vector<size_t> recv_cnts(colors);
407  int result = MPI_Alltoall(
408  &send_cnts[0], 1, mpi_typetraits_u<size_t>::type(), &recv_cnts[0], 1,
409  mpi_typetraits_u<size_t>::type(), MPI_COMM_WORLD);
410 
411  // Start receive operations (non-blocking).
412  std::vector<std::vector<size_t>> rbuffers(colors);
413  std::vector<MPI_Request> requests;
414  for (size_t r(0); r < colors; ++r) {
415  if (recv_cnts[r]) {
416  rbuffers[r].resize(recv_cnts[r]);
417  requests.push_back({});
418  MPI_Irecv(
419  &rbuffers[r][0], recv_cnts[r], mpi_typetraits_u<size_t>::type(), r,
420  0, MPI_COMM_WORLD, &requests[requests.size() - 1]);
421  } // if
422  } // for
423 
424  // Start send operations (blocking is ok here).
425  std::vector<std::vector<size_t>> sbuffers(colors);
426  for (size_t r(0); r < colors; ++r) {
427  if (send_cnts[r]) {
428  std::copy(
429  request_indices[r].begin(), request_indices[r].end(),
430  std::back_inserter(sbuffers[r]));
431 
432  MPI_Send(
433  &sbuffers[r][0], send_cnts[r], mpi_typetraits_u<size_t>::type(), r,
434  0, MPI_COMM_WORLD);
435  } // if
436  } // for
437 
438  // Create a map version of the entity info for lookups below.
439  std::unordered_map<size_t, entity_info_t> entity_info_map;
440  for (auto i : entity_info) {
441  entity_info_map[i.id] = i;
442  } // for
443 
444  // Wait on the receive operations
445  std::vector<MPI_Status> status(requests.size());
446  MPI_Waitall(requests.size(), &requests[0], &status[0]);
447 
448  // Set the offsets for each requested index in the send buffer.
449  for (size_t r(0); r < colors; ++r) {
450  sbuffers[r].resize(rbuffers[r].size());
451 
452  size_t offset(0);
453  for (auto i : rbuffers[r]) {
454  sbuffers[r][offset++] = entity_info_map[i].offset;
455  } // for
456  } // for
457 
458  // we are reusing requests in the following code, need to clear the content
459  // first.
460  requests.clear();
461 
462  // Start receive operations (non-blocking) to get back the
463  // offsets we requested.
464  for (size_t r(0); r < colors; ++r) {
465  // If we sent a request, prepare to receive an answer.
466  if (send_cnts[r]) {
467  // We're done with our receive buffers, so we can re-use them.
468  rbuffers[r].resize(send_cnts[r], 0);
469  requests.push_back({});
470  MPI_Irecv(
471  &rbuffers[r][0], send_cnts[r], mpi_typetraits_u<size_t>::type(), r,
472  0, MPI_COMM_WORLD, &requests[requests.size() - 1]);
473  } // if
474  } // for
475 
476  // Start send operations (blocking is probably ok here).
477  for (size_t r(0); r < colors; ++r) {
478  // If we received a request, prepare to send an answer.
479  if (recv_cnts[r]) {
480  MPI_Send(
481  &sbuffers[r][0], recv_cnts[r], mpi_typetraits_u<size_t>::type(), r,
482  0, MPI_COMM_WORLD);
483  } // if
484  } // for
485 
486  // Wait on the receive operations
487  status.resize(requests.size());
488  MPI_Waitall(requests.size(), &requests[0], &status[0]);
489 
490  std::vector<std::set<size_t>> remote(colors);
491  for (size_t r(0); r < colors; ++r) {
492  for (size_t i(0); i < send_cnts[r]; ++i) {
493  remote[r].insert(rbuffers[r][i]);
494  } // for
495  } // for
496 
497  return remote;
498  } // get_entity_info

626  size_t max_request_indices = 0;
