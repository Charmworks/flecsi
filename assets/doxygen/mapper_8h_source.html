100 void MPIMapper::slice_domain(const Legion::Task *legiontask,
101  const LegionRuntime::HighLevel::Domain &domain,
102  std::vector<DomainSplit> &slices)
103 {
104  // Special cases for startup tasks
105  if (legiontask->task_id == CONNECT_MPI_TASK_ID
106  ||legiontask->task_id ==HANDOFF_TO_MPI_TASK_ID
107  || legiontask->task_id ==WAIT_ON_MPI_TASK_ID)
108  {
109  const int DIM = 2;
110  std::vector<LegionRuntime::HighLevel::Processor> proc_list;
111  std::set<LegionRuntime::HighLevel::Processor> all_procs;
112  machine.get_all_processors(all_procs);
113  for (std::set<LegionRuntime::HighLevel::Processor>::const_iterator it = all_procs.begin();
114  it != all_procs.end(); it++)
115  {
116  if (it->kind() != LegionRuntime::HighLevel::Processor::LOC_PROC)
117  continue;
118  proc_list.push_back(*it);
119  }
120 
121  assert(proc_list.size() == (size_t)(domain.get_rect<DIM>().hi[0]+1));
122  for (LegionRuntime::HighLevel::Domain::DomainPointIterator pir(domain); pir; pir++) {
123  Legion::DomainPoint dp = pir.p;
124  LegionRuntime::Arrays::Point<DIM> p = dp.get_point<DIM>();
125  LegionRuntime::HighLevel::Processor proc = proc_list[p[0]];
126  DomainSplit ds(LegionRuntime::HighLevel::Domain::from_rect<DIM>(Rect<DIM>(p, p)), proc,
127  false , false );
128  slices.push_back(ds);
129  }
130  return;
131  }
132  ShimMapper::decompose_index_space(domain, local_cpus,
133  1/*splitting factor*/, slices);
134 }

190  failed_idx = idx;
191  break;
192  }
193  }
194  log_mapper.error("Failed task mapping for region %d of task %s (%p)\n",
195  failed_idx, legiontask->variants->name, legiontask);
196  assert(false);
197  break;
198  }
199  case Mappable::COPY_MAPPABLE:
200  {
201  Copy *copy = mappable->as_mappable_copy();
202  int failed_idx = -1;
203  for (unsigned idx = 0; idx < copy->src_requirements.size(); idx++)
204  {
205  if (copy->src_requirements[idx].mapping_failed)
206  {
207  failed_idx = idx;
208  break;
209  }
210  }
211  for (unsigned idx = 0; idx < copy->dst_requirements.size(); idx++)
212  {
213  if (copy->dst_requirements[idx].mapping_failed)
214  {
215  failed_idx = copy->src_requirements.size() + idx;
216  break;
217  }
218  }
219  log_mapper.error("Failed copy mapping for region %d of copy (%p)\n",
220  failed_idx, copy);
221  assert(false);
222  break;
223  }
224  default:
225  assert(false);
226  }
