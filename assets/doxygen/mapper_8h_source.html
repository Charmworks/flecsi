210  virtual void slice_task(
211  const Legion::Mapping::MapperContext ctx,
212  const Legion::Task & task,
213  const Legion::Mapping::Mapper::SliceTaskInput & input,
214  Legion::Mapping::Mapper::SliceTaskOutput & output) {
215  using legion_proc = Legion::Processor;
216  context_t & context_ = context_t::instance();
217  if (task.tag == MAPPER_SUBRANK_LAUNCH) {
218  // expect a 1-D index domain
219  assert(input.domain.get_dim() == 1);
220  // send the whole domain to our local processor
221  output.slices.resize(1);
222  output.slices[0].domain = input.domain;
223  output.slices[0].proc = task.target_proc;
224  return;
225  } // end if MAPPER_SUBRANK_LAUNCH
226 
227  if(task.tag ==MAPPER_FORCE_RANK_MATCH){
228  // expect a 1-D index domain - each point goes to the corresponding node
229  assert(input.domain.get_dim() == 1);
230  LegionRuntime::Arrays::Rect<1> r = input.domain.get_rect<1>();
231 
232  // go through all the CPU processors and find a representative for each
233  // node (i.e. address space)
234  std::map<int, Legion::Processor> targets;
235 
236  Legion::Machine::ProcessorQuery pq =
237  Legion::Machine::ProcessorQuery(machine).only_kind(
238  Legion::Processor::LOC_PROC);
239  for(Legion::Machine::ProcessorQuery::iterator it = pq.begin();
240  it != pq.end(); ++it) {
241  Legion::Processor p = *it;
242  int a = p.address_space();
243  if(targets.count(a) == 0)
244  targets[a] = p;
245  }
246 
247  output.slices.resize(r.volume());
248  for(int a = r.lo[0]; a <= r.hi[0]; a++) {
249  assert(targets.count(a) > 0);
250  output.slices[a].domain =
251  Legion::Domain::from_rect<1>(LegionRuntime::Arrays::Rect<1>(a, a));
252  output.slices[a].proc = targets[a];
253  }
254  return;
255  }//MAPPER_FORCE_RANK_MATCH
256 
257  DefaultMapper::slice_task(ctx, task, input, output);
258  // end else
259  }
